\subsection*{Figure S1 — Information-Content Surface: Sensitivity to Signal-to-Noise Ratio}

\begin{figure}[h]
  \centering
  \includegraphics[width=\linewidth]{figures/Figure_1_SI.png}
  \caption{Sensitivity of the information-content surface $I(X;Y)$ to the
    signal-to-noise ratio $\delta/\sigma_{\mathrm{A}}$.
    \textbf{Panel A (left):} $\delta/\sigma_{\mathrm{A}}=0.5$ (weak signal).
    \textbf{Panel B (centre):} $\delta/\sigma_{\mathrm{A}}=1.0$ (nominal case;
    identical parameterisation to \cref{fig:info_surface} in the main text).
    \textbf{Panel C (right):} $\delta/\sigma_{\mathrm{A}}=2.0$ (strong signal).
    All panels share identical colour scaling ($I(X;Y)\in[0,1]$~bits) and axis
    limits.  Sports KPI positions (rugby URC circles, football Championship
    squares) are overlaid using the same two-season segment representation as
    \cref{fig:pef_landscape}.
    The qualitative structure of the surface is robust across this range: the
    high-information region in Quadrants~1 and~2 persists in all three panels,
    though its spatial extent expands markedly with increasing signal strength.
    Generated by \texttt{Matlab/generate\_figure\_S1\_info\_sensitivity.m}.}
  \label{fig:si_info_sensitivity}
\end{figure}

\subsection*{Figure S2 — Labelled KPI Maps: Per-Season Positions (23/24 and 24/25)}

\begin{figure}[h]
  \centering
  \includegraphics[width=\linewidth]{figures/Figure_2_SI.png}
  \caption{Full labelled KPI maps on the PEF landscape for two professional
    sport leagues across seasons 23/24 and 24/25.
    \textbf{Panel A (left):} All rugby union KPIs from the United Rugby
    Championship (URC).
    \textbf{Panel B (right):} All association-football KPIs from the English
    Championship.
    Open markers denote the season~1 (23/24) position; filled markers denote
    the season~2 (24/25) position; arrows indicate the direction of
    year-on-year change.  KPIs whose $(\kappa,\rho)$ position migrates from
    one quadrant to another between seasons are highlighted in vermillion;
    all others retain the sport category colour (blue: rugby; orange:
    football).  Quadrant boundaries ($\rho=0$ and $\kappa=1$) are drawn in
    black.  Individual KPI labels are placed at the season-2 position.
    An unlabelled summary of these data appears in
    \cref{fig:pef_landscape} of the main text, where each KPI is represented
    by a two-season mean marker and a connecting segment.  Generated by
    \texttt{Matlab/generate\_figure\_S2\_labelled\_kpis.m}.}
  \label{fig:si_kpi_labelled}
\end{figure}
