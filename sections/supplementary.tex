\subsection*{Figure S4 --- Idealised Probit Simulation: Information vs.\ ML Gain (Stratified by $\delta/\sigma_{\mathrm{A}}$)}

\begin{figure}[h]
  \centering
  \includegraphics[width=\linewidth]{figures/Figure_S4_idealised_I_vs_dML_stratified.png}
  \caption{Theory-aligned idealised probit simulation under assumptions
    (A1)--(A2): bivariate-normal $(X_{\mathrm{A}},X_{\mathrm{B}})$ on a
    $4\times4\times4$ grid of $(\kappa,\rho,\delta/\sigma_{\mathrm{A}})$
    with $\kappa\in\{0.8,1.0,1.2,2.0\}$,
    $\rho\in\{-0.3,-0.15,0,0.4\}$, and
    $\delta/\sigma_{\mathrm{A}}\in\{0.3,0.5,1.0,2.0\}$ (64 admissible
    cells; $n=5{,}000$ per cell, 50 Monte Carlo trials, five-fold logistic
    CV matching the empirical protocol).
    Each panel fixes $\delta/\sigma_{\mathrm{A}}$ and plots analytic
    $I(X;Y)$ against mean ML improvement (\%).
    Pooled across all cells,
    $r=-0.71$ for $I$ vs.\ $\Delta\mathrm{ML}$ (misleading when $\delta$ varies).
    Stratified: $r=+0.39$ at $\delta/\sigma_{\mathrm{A}}=0.3$,
    $+0.31$ at $0.5$, $-0.18$ at $1.0$, $-0.87$ at $2.0$.
    The $\delta/\sigma_{\mathrm{A}}=0.3$ slice is nearest the median
    empirical KPI value ($0.28$; Figure~S5).
    All 16 Quadrant~4 cells with $\eta<1$ show positive mean ML gain.
    Generated by \texttt{scripts/paper\_pipeline/run\_pef\_idealised\_probit\_sim.m}
    and \texttt{run\_pef\_finalize\_diagnostics.m}.}
  \label{fig:si_idealised_stratified}
\end{figure}
